def plt_nlargest_pie(df, column, index, title="", top=10):
    data = df.nlargest(top, columns=[column]).set_index(index)  
    data.plot(kind='pie', y=column,autopct='%1.0f%%', figsize=(10, 5), legend=False).set(label=data.index)
    plt.title(title)
    plt.ylabel("")
    plt.show()
